DECLARE
	a POSITIVE(3):=444;
	b NUMBER(3):=222;
	c CHAR:='c';
	d BOOLEAN:=NOT TRUE;
BEGIN
	WHILE a>b LOOP
		BEGIN
			a:=a-50;
			DBMS_OUTPUT.PUT(a);
			DBMS_OUTPUT.NEW_LINE;
			b:=8;
			DBMS_OUTPUT.PUT(a);
			DBMS_OUTPUT.NEW_LINE;
			d:=TRUE;
		END;
	END LOOP;
	a:=NUMBER(2)(a);
	IF BEGIN a<b THEN
		DBMS_OUTPUT.PUT(a);
	END IF;
END;\
